import groovy.json.JsonSlurper
node {
   
    stage('fetch') { 
        URL todo_url = new URL("https://jsonplaceholder.typicode.com/todos");
       
         def todo_list = new JsonSlurper().parse(todo_url)
        //  println(card)
        for(def todo : todo_list) {
            if(todo.completed == false) {
                println(todo)
            }
        }
    }
  
  
}
